-- 1. Tạo database với UTF8MB4 để hỗ trợ tiếng Việt có dấu
CREATE DATABASE quanlylinkkien 
    CHARACTER SET utf8mb4 
    COLLATE utf8mb4_unicode_ci;
USE quanlylinkkien;

-- 2. Tạo bảng LOAIHANG
CREATE TABLE LOAIHANG (
    MALOAIHANG INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    TENLOAIHANG VARCHAR(100) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 3. Tạo bảng THUONGHIEU
CREATE TABLE THUONGHIEU (
    MATHUONGHIEU INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    TENTHUONGHIEU VARCHAR(100) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 4. Tạo bảng MATHANG
CREATE TABLE MATHANG (
    MAHANG INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    TENHANG VARCHAR(100) NOT NULL UNIQUE,
    MALOAIHANG INT NOT NULL,
    MATHUONGHIEU INT NOT NULL,
    DONVITINH VARCHAR(20) NOT NULL,
    GIABAN DECIMAL(19,4) NOT NULL,
    ANHMINHHOA VARCHAR(200),
    THONGSOKYTHUAT TEXT,
    FOREIGN KEY (MALOAIHANG) REFERENCES LOAIHANG(MALOAIHANG),
    FOREIGN KEY (MATHUONGHIEU) REFERENCES THUONGHIEU(MATHUONGHIEU)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 5. Tạo bảng KHACHHANG
CREATE TABLE KHACHHANG (
    MAKHACHHANG INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    HOKHACHHANG VARCHAR(100) NOT NULL,
    TENKHACHHANG VARCHAR(100) NOT NULL,
    GIOITINH VARCHAR(10) NOT NULL,
    DIACHI VARCHAR(100) NOT NULL,
    EMAIL VARCHAR(50),
    DIENTHOAI VARCHAR(20) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 6. Tạo bảng DONHANG
CREATE TABLE DONHANG (
    MADONHANG INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    MAKHACHHANG INT NOT NULL,
    NGAYBANHANG DATETIME,
    FOREIGN KEY (MAKHACHHANG) REFERENCES KHACHHANG(MAKHACHHANG)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 7. Tạo bảng ctDONHANG (Chi tiết đơn hàng)
CREATE TABLE ctDONHANG (
    MADONHANG INT NOT NULL,
    MAHANG INT NOT NULL,
    GIABAN DECIMAL(19,4) NOT NULL,
    SOLUONG DECIMAL(10,2) NOT NULL,
    MUCGIAMGIA DECIMAL(4,2) NOT NULL DEFAULT 0,
    PRIMARY KEY (MADONHANG, MAHANG),
    FOREIGN KEY (MADONHANG) REFERENCES DONHANG(MADONHANG),
    FOREIGN KEY (MAHANG) REFERENCES MATHANG(MAHANG)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
